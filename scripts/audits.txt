CREATE TABLE audit_lotesnwlinha (audit_id INT AUTO_INCREMENT PRIMARY KEY,`audit_timestamp` DATETIME, `action` varchar(20)) SELECT * FROM lotesnwlinha;
DELIMITER //
CREATE TRIGGER `lotesnwlinha_BEFORE_UPDATE` BEFORE UPDATE ON `lotesnwlinha` 
FOR EACH ROW INSERT INTO audit_lotesnwlinha 
SELECT NULL, NOW(),'update', t.* FROM lotesnwlinha t WHERE id = OLD.id
//
DELIMITER ;

DELIMITER //
CREATE TRIGGER `lotesnwlinha_BEFORE_DELETE` BEFORE DELETE ON `lotesnwlinha` 
FOR EACH ROW INSERT INTO audit_lotesnwlinha 
SELECT NULL, NOW(),'delete', t.* FROM lotesnwlinha t WHERE id = OLD.id
//
DELIMITER ;
